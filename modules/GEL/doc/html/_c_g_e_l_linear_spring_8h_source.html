119 inline void computeForces()
120 {
121 // update basic parameters of current link
122 chai3d::cVector3d m_link01 = cSub(m_node1->m_pos, m_node0->m_pos);
123 double m_length = m_link01.length();
124
125 //-------------------------------------------------------------
126 // ELONGATION:
127 //-------------------------------------------------------------
128 // if distance too small, no forces are applied
129 if (m_length < 0.000001) { return; }
130
131 // elongation compute force
132 double f = m_kSpringElongation * (m_length - m_length0);
133
134 // apply force
135 if (m_length > 0.000001)
136 {
137 chai3d::cVector3d force = cMul(f/m_length, m_link01);
138 m_node0->addForce(force);
139 chai3d::cVector3d tmpfrc = cMul(-1, force);
140 m_node1->addForce(tmpfrc);
141 }
142 }
